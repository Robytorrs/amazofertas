  <script>
    // Ejemplo de datos de prueba
    const ofertas = [
      {
        titulo: "Auriculares Bluetooth",
        precio: "29.99",
        imagen: "https://via.placeholder.com/250x160?text=Auriculares",
        enlace: "https://www.amazon.es/"
      },
      {
        titulo: "Teclado Mecánico",
        precio: "59.99",
        imagen: "https://via.placeholder.com/250x160?text=Teclado",
        enlace: "https://www.amazon.es/"
      }
    ];

    function crearOfertaHTML(oferta, index) {
      const div = document.createElement('div');
      div.className = 'oferta-card';
      div.innerHTML = `
        <img src="${oferta.imagen || 'https://via.placeholder.com/250x160?text=Sin+Imagen'}" alt="Imagen oferta" />
        <h3>${oferta.titulo}</h3>
        <div class="precio">${oferta.precio ? '€ ' + oferta.precio : 'Precio no disponible'}</div>
        <div class="btns">
          <a href="${oferta.enlace}" target="_blank" rel="noopener noreferrer">Ver en Amazon</a>
          <button onclick="eliminarOferta(${index})">Eliminar</button>
        </div>
      `;
      return div;
    }

    function mostrarOfertas() {
      const contenedor = document.getElementById('lista-ofertas');
      contenedor.innerHTML = '';
      ofertas.forEach((oferta, index) => {
        contenedor.appendChild(crearOfertaHTML(oferta, index));
      });
    }

    function eliminarOferta(index) {
      ofertas.splice(index, 1);
      mostrarOfertas();
    }

    mostrarOfertas();
  </script>
